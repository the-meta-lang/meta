[defunc [strlen string]
	[define length 0]
	[while [!= &1[+ string length] 0]
		[set! length [+ length 1]]]
	[return length]]

[defunc [strcat string1 string2]
	[define length1 [strlen string1]]
	[define length2 [strlen string2]]
	[define newstring [malloc [+ length1 length2]]]
	[define i 0]
	[while [!= i length1]
		[set! newstring [+ string1 i]]
		[set! i [+ i 1]]]
	[while [!= i length2]
		[set! newstring [+ string2 i]]
		[set! i [+ i 1]]]]
